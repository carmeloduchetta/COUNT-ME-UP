CREATE KEYSPACE bbc_stage WITH replication = {
  'class': 'SimpleStrategy',
  'replication_factor': '1'
};

USE bbc_stage;

CREATE TABLE "User" (
  key text,
  "___class" text,
  company text,
  ddi text,
  email text,
  firstname text,
  "isActive" text,
  "isCandidate" text,
  lastname text,
  "mainTel" text,
  mobile text,
  name text,
  notes text,
  password text,
  telephone2 text,
  title text,
  username text,
  PRIMARY KEY (key)
) WITH COMPACT STORAGE AND
  bloom_filter_fp_chance=0.010000 AND
  caching='KEYS_ONLY' AND
  comment='' AND
  dclocal_read_repair_chance=0.000000 AND
  gc_grace_seconds=864000 AND
  index_interval=128 AND
  read_repair_chance=1.000000 AND
  replicate_on_write='true' AND
  populate_io_cache_on_flush='false' AND
  default_time_to_live=0 AND
  speculative_retry='NONE' AND
  memtable_flush_period_in_ms=0 AND
  compaction={'class': 'SizeTieredCompactionStrategy'} AND
  compression={'sstable_compression': 'LZ4Compressor'};

CREATE INDEX bbc_stage_User____class ON "User" ("___class");

CREATE INDEX bbc_stage_User_firstname ON "User" (firstname);

CREATE INDEX bbc_stage_User_isActive ON "User" ("isActive");

CREATE INDEX bbc_stage_User_lastname ON "User" (lastname);

  comment='' AND

  gc_grace_seconds=864000 AND
  index_interval=128 AND
  read_repair_chance=1.000000 AND
  replicate_on_write='true' AND
  populate_io_cache_on_flush='false' AND
  default_time_to_live=0 AND
  speculative_retry='NONE' AND
  memtable_flush_period_in_ms=0 AND

  compression={'sstable_compression': 'LZ4Compressor'};
